function fibonacci(n) {
  if (n === 1) {
    return [0, 1];
  } else {
    var s = fibonacci(n - 1);
    s.push(s[s.length - 1] + s[s.length - 2]);
    return s;
  }
}
function isPrime(num) {
  for (let i = 2; i < num; i++) if (num % i === 0) return false;
  return num !== 1 && num !== 0;
}
function isEven(n) {

function isOdd(n) {
  return Math.abs(n % 2) == 1;
}
